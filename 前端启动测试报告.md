
2. **配置后端**
   ```bash
   cd unified-backend
   cp .env.example .env
   # 编辑 .env 文件，配置数据库连接
   npm start
   ```

3. **启动前端**
   ```bash
   cd unified-frontend
   npm install
   npm run dev
   ```

   - 前端: http://localhost:5173
